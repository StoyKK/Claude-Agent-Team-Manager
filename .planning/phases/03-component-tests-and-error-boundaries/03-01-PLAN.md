
2. Add `unhandledrejection` listener in a `useEffect` inside the `App` function (per D-05). Place it BEFORE the existing keyboard shortcut useEffect:
   ```typescript
   useEffect(() => {
     function handleRejection(event: PromiseRejectionEvent) {
       const message = event.reason instanceof Error
         ? event.reason.message
         : String(event.reason);
       logger.error("unhandled-rejection", message);
       toast("An unexpected error occurred", "error");
     }
     window.addEventListener("unhandledrejection", handleRejection);
     return () => window.removeEventListener("unhandledrejection", handleRejection);
   }, []);
   ```

3. Wrap the `<SetupWizard />` in the return JSX with:
   ```tsx
   <ErrorBoundary fallbackRender={ErrorFallback}>
     <SetupWizard />
   </ErrorBoundary>
   ```

   Keep the `<div className="tree-panel">` outside the boundary so layout is preserved.
