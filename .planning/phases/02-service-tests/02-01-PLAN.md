The final structure must be:

```typescript
import { defineConfig, mergeConfig } from "vite";
import react from "@vitejs/plugin-react";
import path from "path";

const host = process.env.TAURI_DEV_HOST;

export default defineConfig(async () =>
  mergeConfig(
    {
      plugins: [react()],
      resolve: {
        alias: {
          "@": path.resolve(__dirname, "./src"),
          fs: path.resolve(__dirname, "./src/utils/fs-stub.ts"),
        },
      },
      clearScreen: false,
      server: {
        port: 5173,
        strictPort: true,
        host: host || false,
        hmr: host
          ? {
              protocol: "ws",
              host,
              port: 1421,
            }
          : undefined,
        watch: {
          ignored: ["**/src-tauri/**"],
        },
      },
    },
    {
      test: {
        environment: "jsdom",
        setupFiles: ["./src/vitest.setup.ts"],
        globals: true,
      },
    }
  )
);
```

- `mergeConfig` imported alongside `defineConfig` from `"vite"`
- Return value of async factory is `mergeConfig(baseConfig, testConfig)` — NOT a nested object
- `environment: "jsdom"` per D-04
- `globals: true` enables `describe`, `it`, `expect`, `vi` without imports
- `setupFiles` points to the new setup file
- All existing config (plugins, resolve.alias, server) preserved exactly
